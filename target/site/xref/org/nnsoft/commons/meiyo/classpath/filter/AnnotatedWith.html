16  package org.nnsoft.commons.meiyo.classpath.filter;
17  
18  import java.lang.annotation.Annotation;
19  
20  /**
21   * A filter that verifies the class found is annotated with the given annotation.
22   *
23   * @version $Id$
24   */
25  final class AnnotatedWith implements Filter {
26  
27      /**
28       * The annotation has to be searched on classes.
29       */
30      private final Annotation annotation;
31  
32      /**
33       * Crates a new annotation filter.
34       *
35       * @param annotation the annotation has to be searched on classes.
36       */
37      public AnnotatedWith(Annotation annotation) {
38          this.annotation = annotation;
39      }
40  
41      /**
42       * {@inheritDoc}
43       */
44      public boolean matches(Class<?> clazz) {
45          Annotation fromElement = clazz.getAnnotation(this.annotation.annotationType());
46          return this.annotation.equals(fromElement);
47      }
48  
49      /**
50       * {@inheritDoc}
51       */
52      @Override
53      public String toString() {
54          return "annotatedWith("
55                  + this.annotation
56                  + ")";
57      }
58  
59  }
