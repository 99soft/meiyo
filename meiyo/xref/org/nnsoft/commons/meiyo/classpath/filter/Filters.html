36      public static Filter and(Filter...filters) {
37          if (filters == null) {
38              throw new IllegalArgumentException("Parameter 'filters' must be not null");
39          }
40          if (filters.length == 0) {
41              throw new IllegalArgumentException("Empty filters list not allowed");
42          }
43  
44          return new And(filters);
45      }

47      public static Filter annotatedWith(Annotation annotation) {
48          if (annotation == null) {
49              throw new IllegalArgumentException("Parameter 'annotation' must not be null");
50          }
51  
52          checkForRuntimeRetention(annotation.annotationType());
53  
54          return new AnnotatedWith(annotation);
55      }
56  
57      public static Filter annotatedWithType(Class<? extends Annotation> annotationType) {
58          if (annotationType == null) {
59              throw new IllegalArgumentException("Parameter 'annotationType' must not be null");
60          }
61  
62          checkForRuntimeRetention(annotationType);
63  
64          return new AnnotatedWithType(annotationType);
65      }
66  
67      private static void checkForRuntimeRetention(Class<? extends Annotation> annotationType) {
68          Retention retention = annotationType.getAnnotation(Retention.class);
69          if (retention == null || RetentionPolicy.RUNTIME != retention.value()) {
70              throw new IllegalArgumentException("Annotation @"
71                      + annotationType.getName()
72                      + " is missing RUNTIME retention");
73          }
74      }
